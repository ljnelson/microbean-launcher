072/**

074 * artifact coordinates into a {@link Set} of classpath {@link URI}s

077 * @author <a href="https://about.me/lairdnelson"
078 * target="_parent">Laird Nelson</a>

080 * @see #getClasspath()
081 *
082 * @see #getRemainingCommandLineArguments()
083 */
084@Singleton
085public final class ClasspathExporter {

149  /**
150   * A <a
151   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#producer_method"
152   * target="_parent">producer method</a> that {@linkplain Produces
153   * produces} an {@link ApplicationScoped application-scoped} {@link
154   * List} of {@link Dependency} instances that represent Maven-style
155   * artifact coordinates from either {@linkplain
156   * #setDependencies(List) explicitly-set <code>Dependency</code>
157   * instances} or {@linkplain CommandLine command-line arguments} or
158   * system properties.
159   *
160   * <p>This method never returns {@code null}.</p>
161   *
162   * @param commandLine a {@link CommandLine} instance; may be {@code
163   * null} in which case an {@linkplain Collection#isEmpty() empty}
164   * {@link List} will be returned
165   *
166   * @return a non-{@code null} {@link List} of {@link Dependency}
167   * instances
168   *
169   * @see #setDependencies(List)
170   *
171   * @see #getDependencies()
172   */
173  @Produces
174  @ApplicationScoped
175  public final List<Dependency> produceDependencies(final CommandLine commandLine) {
176    final List<Dependency> returnValue;
177    List<Dependency> dependencies = this.getDependencies();
178    if (dependencies == null) {
179      returnValue = new ArrayList<>();
180      if (commandLine != null) {
181        final String[] artifactPath;
182        if (commandLine.hasOption("artifactPath")) {
183          artifactPath = commandLine.getOptionValues("artifactPath");
184        } else {
185          final String artifactPathSystemProperty = System.getProperty("maven.artifact.path");
186          if (artifactPathSystemProperty == null) {
187            artifactPath = new String[0];
188          } else {
189            artifactPath = new String[] { artifactPathSystemProperty };
190          }
191        }
192        if (artifactPath != null) {
193          Arrays.stream(artifactPath)
194            .flatMap(item -> Arrays.stream(item.split("[, ]+")))
195            .forEach(gav -> {
196                if (gav != null) {
197                  final String[] components = gav.split(":");
198                  assert components != null;
199                  assert components.length > 0;
200                  String groupId = null;
201                  String artifactId = null;
202                  String version = null;
203                  String classifier = null;
204                  String packaging = null;
205                  String scope = null;
206                  switch (components.length) {
207                  case 1:
208                    // artifact
209                    groupId = null; // will be set to default later
210                    artifactId = components[0];
211                    version = "LATEST";
212                    packaging = "jar";
213                    scope = JavaScopes.COMPILE;
214                    break;
215                  case 2:
216                    // group:artifact or artifact:version
217                    final String firstComponent = components[0];
218                    assert firstComponent != null;
219                    final String secondComponent = components[1];
220                    assert secondComponent != null;
221                    if (secondComponent.isEmpty()) {
222                      throw new IllegalArgumentException("Unrecognized artifact coordinates: " + gav);
223                    } else if (Character.isDigit(secondComponent.charAt(0))) {
224                      // artifact:version
225                      groupId = null; // will default below
226                      artifactId = firstComponent;
227                      version = secondComponent;
228                    } else {
229                      // group:artifact
230                      groupId = firstComponent;
231                      artifactId = secondComponent;
232                      version = "LATEST";
233                    }
234                    packaging = "jar";
235                    scope = JavaScopes.COMPILE;
236                    break;
237                  case 3:
238                    // group:artifact:version
239                    groupId = components[0];
240                    artifactId = components[1];
241                    version = components[2];
242                    packaging = "jar";
243                    scope = JavaScopes.COMPILE;
244                    break;
245                  case 4:
246                    // group:artifact:version:packaging
247                    groupId = components[0];
248                    artifactId = components[1];
249                    version = components[2];
250                    packaging = components[3];
251                    scope = JavaScopes.COMPILE;
252                    break;
253                  case 5:
254                    // group:artifact:version:packaging:classifier
255                    groupId = components[0];
256                    artifactId = components[1];
257                    version = components[2];
258                    packaging = components[3];
259                    classifier = components[4];
260                    scope = JavaScopes.COMPILE;
261                    break;
262                  case 6:
263                    // group:artifact:version:packaging:classifier:scope
264                    groupId = components[0];
265                    artifactId = components[1];
266                    version = components[2];
267                    packaging = components[3];
268                    classifier = components[4];
269                    scope = components[5];
270                    break;
271                  default:
272                    throw new IllegalArgumentException("Unrecognized artifact coordinates: " + gav);
273                  }
274                  if (groupId == null || groupId.isEmpty()) {
275                    groupId = commandLine.getOptionValue("defaultGroupId");
276                  }
277                  if (groupId == null || groupId.isEmpty() ||
278                      artifactId == null || artifactId.isEmpty()) {
279                    throw new IllegalArgumentException("Unrecognized artifact coordinates: " + gav);
280                  }
281                  if (version == null || version.isEmpty()) {
282                    version = "LATEST";
283                  }
284                  // note: classifier can be null
285                  if (packaging == null || packaging.isEmpty()) {
286                    packaging = "jar";
287                  }
288                  if (scope == null || scope.isEmpty()) {
289                    scope = JavaScopes.COMPILE;
290                  }
291                  assert scope != null;
292                  returnValue.add(new Dependency(new DefaultArtifact(groupId, artifactId, classifier, packaging, version), scope));
293                }
294              });
295        }
296        this.remainingCommandLineArguments = commandLine.getArgs();
297      }
298      this.dependencies = returnValue;
299    } else {
300      returnValue = dependencies;
301    }
302    if (returnValue == null || returnValue.isEmpty()) {
303      return Collections.emptyList();
304    } else {
305      return Collections.unmodifiableList(returnValue);
306    }
307  }

309  /**
310   * Returns the {@link List} of {@link Dependency} instances that
311   * will be used by this class to compute the eventual classpath
312   * returned by the {@link #getClasspath()} method.
313   *
314   * <p>This method may return {@code null}.</p>
315   *
316   * @return a {@link List} of {@link Dependency} instances, or {@code
317   * null}
318   *
319   * @see #setDependencies(List)
320   */
321  public final List<Dependency> getDependencies() {
322    final List<Dependency> returnValue;
323    if (this.dependencies == null) {
324      returnValue = null;
325    } else if (this.dependencies.isEmpty()) {
326      returnValue = Collections.emptyList();
327    } else {
328      returnValue = Collections.unmodifiableList(this.dependencies);
329    }
330    return returnValue;
331  }

333  /**
334   * Sets the {@link List} of {@link Dependency} instances that will
335   * be used by this {@link ClasspathExporter} to compute the
336   * classpath returned by the {@link #getClasspath()} method.
337   *
338   * <p><strong>Users of this class should not feel obligated to call
339   * this method.</strong> If this method is not explicitly called,
340   * this class will infer the list of dependencies from command line
341   * arguments and system properties.</p>
342   *
343   * @param dependencies a {@link List} of {@link Dependency}
344   * instances; may be {@code null}
345   *
346   * @see #getDependencies()
347   *
348   * @see #getClasspath()
349   */
350  public final void setDependencies(final List<Dependency> dependencies) {
351    if (dependencies == null || dependencies.isEmpty()) {
352      this.dependencies = Collections.emptyList();
353    } else {
354      this.dependencies = new ArrayList<>(dependencies);
355    }
356  }

358  /**
359   * Given a {@link List} of {@link Dependency} elements supplied as
360   * the value of the {@code dependencies} parameter, uses the
361   * supplied {@link RepositorySystem} to {@linkplain
362   * RepositorySystem#resolveDependencies(RepositorySystemSession,
363   * DependencyRequest) resolve} them in {@linkplain
364   * JavaScopes#COMPILE compile scope}, and uses their {@linkplain
365   * Artifact#getFile() associated <code>File</code> instances} to
366   * build a {@linkplain #getClasspath() classpath} out of them.
367   *
368   * <p>The classpath so built will consist of {@link URI}s that
369   * represent {@link File}s on the local filesystem.</p>
370   *
371   * @param event the event signalling container startup; ignored; may
372   * be {@code null}
373   *
374   * @param repositorySystem the {@link RepositorySystem} used for
375   * dependency resolution; must not be {@code null}
376   *
377   * @param session the {@link RepositorySystemSession} used to
378   * interact with the supplied {@code repositorySystem}; must not be
379   * {@code null}
380   *
381   * @param remoteRepositories a {@link List} of {@link
382   * RemoteRepository} instances to use for {@linkplain Resolution
383   * dependency resolution} (as opposed to, say, deployment or some
384   * other repository operation); may be {@code null} but probably
385   * shouldn't be
386   *
387   * @param dependencies a {@link List} of {@link Dependency}
388   * instances to {@linkplain
389   * RepositorySystem#resolveDependencies(RepositorySystemSession,
390   * DependencyRequest) resolve}; may be {@code null} in which case no
391   * action will be taken and the {@link #getClasspath()} method will
392   * return an {@linkplain Collections#emptySet() empty
393   * <code>Set</code>} of {@link URI}s
394   *
395   * @exception NullPointerException if {@code repositorySystem} or
396   * {@code session} is {@code null}
397   *
398   * @exception DependencyResolutionException if there was a problem
399   * with {@linkplain
400   * RepositorySystem#resolveDependencies(RepositorySystemSession,
401   * DependencyRequest) dependency resolution};
402   *
403   * @see #getClasspath()
404   *
405   * @see
406   * RepositorySystem#resolveDependencies(RepositorySystemSession,
407   * DependencyRequest)
408   */
409  private final void onStartup(@Observes @Initialized(ApplicationScoped.class) final Object event,
410                               final RepositorySystem repositorySystem,
411                               final RepositorySystemSession session,
412                               @Resolution final List<RemoteRepository> remoteRepositories,
413                               final List<Dependency> dependencies)
414  throws DependencyResolutionException {
415    Objects.requireNonNull(repositorySystem);
416    Objects.requireNonNull(session);
417    if (dependencies != null && !dependencies.isEmpty()) {
418      final CollectRequest collectRequest = new CollectRequest((Dependency)null /* no root */, dependencies, remoteRepositories);
419      final DependencyRequest dependencyRequest = new DependencyRequest(collectRequest, DependencyFilterUtils.classpathFilter(JavaScopes.COMPILE)); // TODO: not sure about this filter if the user can specify individual scopes
420      final DependencyResult dependencyResult = repositorySystem.resolveDependencies(session, dependencyRequest);
421      assert dependencyResult != null;
422      final List<ArtifactResult> artifactResults = dependencyResult.getArtifactResults();
423      assert artifactResults != null;
424      for (final ArtifactResult artifactResult : artifactResults) {
425        if (artifactResult != null) {
426          final Artifact resolvedArtifact = artifactResult.getArtifact();
427          if (resolvedArtifact != null) {
428            final File f = resolvedArtifact.getFile();
429            assert f != null;
430            assert f.isFile();
431            assert f.canRead();
432            final URI uri = f.toURI();
433            assert uri != null;
434            this.classpath.add(uri);
435          }
436        }
437      }
438    }
439  }

441  /**
442   * Returns an {@linkplain Collections#unmodifiableSet(Set)
443   * unmodifiable <code>Set</code>} of {@link URI}s representing Java
444   * classpath elements that resulted from examining the user-supplied
445   * <em>artifact path</em>&mdash;a list of Maven-style artifact
446   * coordinates.
447   *
448   * <p>This method never returns {@code null}.</p>
449   *
450   * @return a non-{@code null}, {@linkplain
451   * Collections#unmodifiableSet(Set) unmodifiable <code>Set</code>}
452   * of {@link URI}s representing Java classpath elements
453   */
454  public final Set<URI> getClasspath() {
455    if (this.classpath == null || this.classpath.isEmpty()) {
456      return Collections.emptySet();
457    } else {
458      return Collections.unmodifiableSet(this.classpath);
459    }
460  }

462  /**
463   * Returns a {@link String} array representing any command line
464   * arguments that this class did <em>not</em> "consume" as it tried
465   * to determine the user-supplied artifact path.
466   *
467   * <p>This method will never return {@code null}.</p>
468   *
469   * @return a non-{@code null} {@link String} array representing any
470   * command line arguments that this class did <em>not</em> "consume"
471   * as it tried to determine the user-supplied artifact path
472   */
473  public final String[] getRemainingCommandLineArguments() {
474    if (this.remainingCommandLineArguments == null) {
475      return new String[0];
476    } else {
477      return this.remainingCommandLineArguments.clone();
478    }
479  }

487  /**
488   * A <a
489   * href="http://docs.jboss.org/cdi/spec/2.0/cdi-spec.html#producer_method"
490   * target="_parent">producer method</a> that {@linkplain Produces
491   * produces} an {@link Options} instance representing the command
492   * line options that this class looks for in its {@link
493   * #produceDependencies(CommandLine)} method.
494   *
495   * <p>This method never returns {@code null}.</p>
496   *
497   * @return a non-{@code null} {@link Options} representing the
498   * command line options this class will look for in its {@link
499   * #produceDependencies(CommandLine)} method
500   *
501   * @see #produceDependencies(CommandLine)
502   */
503  @Produces
504  @ApplicationScoped
505  private static final Options getOptions() {
506    final Options options = new Options();
507
508    final Option defaultGroupId = Option.builder()
509      .longOpt("defaultGroupId")
510      .hasArg(true)
511      .required(false)
512      .type(String.class)
513      .argName("groupId")
514      .desc("The default groupId to use in artifact coordinates.")
515      .build();
516    options.addOption(defaultGroupId);
517
518    final Option artifact = Option.builder("ap")
519      .longOpt("artifactPath")
520      .hasArgs()
521      .required(false)
522      .type(String.class)
523      .argName("GAV coordinates")
524      .desc("The double-colon separated GAV coordinates of the artifacts to resolve.")
525      .build();
526    options.addOption(artifact);
527
528    return options;
529  }
