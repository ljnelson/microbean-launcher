124  /**
125   * {@linkplain SeContainerInitializer#initialize() Starts} a
126   * {@linkplain SeContainer CDI container} with a classpath formed
127   * from the existing classpath and the local filesystem locations of
128   * Maven artifacts and their transitive dependencies resolvable from
129   * the value of the {@code --artifactPath} command line option (or
130   * its short form, {@code -ap}).
131   *
132   * <p>The supplied {@code bootstrapInitializer} is used to
133   * {@linkplain SeContainerInitializer#initialize() start} a
134   * {@linkplain SeContainer CDI container} that indirectly uses
135   * components from the <a
136   * href="https://ljnelson.github.io/microbean-maven-cdi/">MicroBean
137   * Maven CDI</a>, <a
138   * href="https://ljnelson.github.io/microbean-main/">MicroBean
139   * Main</a> and <a
140   * href="https://ljnelson.github.io/microbean-commons-cli/">MicroBean
141   * Commons CLI</a> projects to parse the value of the {@code
142   * --artifactPath} command line option and, by invoking the <a
143   * href="https://ljnelson.github.io/microbean-maven-cdi/">MicroBean
144   * Maven CDI</a> machinery, <a
145   * href="https://maven.apache.org/resolver/maven-resolver-api/apidocs/org/eclipse/aether/RepositorySystem.html#resolveDependencies(org.eclipse.aether.RepositorySystemSession,%20org.eclipse.aether.resolution.DependencyRequest)">resolves
146   * each such artifact mentioned, together with its transitive
147   * compile- and runtime-scoped dependencies</a> into the user's <a
148   * href="https://maven.apache.org/guides/introduction/introduction-to-repositories.html">local
149   * Maven repository</a>.  A classpath is built from the (local
150   * filesystem) locations of these artifacts and the current
151   * classpath.</p>
152   *
153   * <p>The {@code --artifactPath} command line option (or its short
154   * form, {@code -ap}) takes a whitespace- and/or comma-separated
155   * list of Maven artifact coordinates of one of the following
156   * forms:</p>
157   *
158   * <ul>
159   *
160   * <li>{@code groupId:artifactId:version:packaging:classifier:scope}</li>
161   *
162   * <li>{@code groupId:artifactId:version:packaging:classifier}</li>
163   *
164   * <li>{@code groupId:artifactId:version:packaging}</li>
165   *
166   * <li>{@code groupId:artifactId:version}</li>
167   *
168   * <li>{@code groupId:artifactId}</li>
169   *
170   * <li>{@code artifactId:version} (requires the {@code
171   * --defaultGroupId} command line option to be specified; see
172   * below)</li>
173   *
174   * </ul>
175   *
176   * <p>In all cases, {@code packaging} is the kind of artifact
177   * desired, e.g. {@code jar}, {@code war}, {@code ear}, etc., and
178   * usually translates to a three-character file extension suffix.
179   * It defaults to {@code jar}.  {@code classifier} is <a
180   * href="https://maven.apache.org/pom.html#Dependencies">Maven's
181   * notion of a classifier</a>&mdash;an identifier to further
182   * classify an artifact, and is usually not specified.  {@code
183   * scope} is one of {@code compile}, {@code runtime}, or {@code
184   * test} and is {@code compile} by default.</p>
185   *
186   * <p>If a {@code --defaultGroupId} command line option is present,
187   * then its value is taken to be a <a
188   * href="https://maven.apache.org/pom.html#Dependencies">Maven group
189   * identifier</a>, and elements in the artifact path that have that
190   * group identifier may be specified as simply {@code
191   * artifactId:version}.</p>
192   *
193   * <p>Examples of valid command line options described here include:</p>
194   *
195   * <ul>
196   *
197   * <li>The most common style of specification:
198   * {@code --artifactPath com.foobar:frobnicator:1.0,com.foobar:caturgiator:2.0}</li>
199   *
200   * <li>Using {@code --defaultGroupId}:
201   * {@code --defaultGroupId com.foobar -ap frobnicator:1.0,caturgiator:2.0}</li>
202   *
203   * <li>Maximally specific (note the double-colon eliminating the
204   * {@code classifier} component):
205   * {@code --artifactPath com.foobar:frobnicator:1.0:jar::compile,com.foobar:caturgiator:2.0:jar::compile}</li>
206   *
207   * </ul>
208   *
209   * <p>The "bootstrap" dependency resolution CDI container is then
210   * {@linkplain SeContainer#close() closed} and discarded.</p>
211   *
212   * <p>The classpath constructed according to the prior description
213   * is the classpath used by the supplied {@code initializer}, which
214   * is then used to {@linkplain SeContainerInitializer#initialize()
215   * start} the "real" {@linkplain SeContainer CDI container}.</p>
216   *
217   * <p>If there is a {@code consumer}, then the {@link SeContainer}
218   * so started is supplied to it.</p>
219   *
220   * <p>Finally the {@link SeContainer} is {@linkplain
221   * SeContainer#close() closed} and this method returns.
222   *
223   * @param bootstrapInitializer the {@link SeContainerInitializer} to
224   * use while performing dependency resolution; may be {@code null}
225   * in which case the return value of {@link
226   * SeContainerInitializer#newInstance()} will be used instead
227   *
228   * @param initializer the {@link SeContainerInitializer} to use to
229   * {@linkplain SeContainerInitializer#initialize() create and run}
230   * the "real" CDI container; may be {@code null}
231   * in which case the return value of {@link
232   * SeContainerInitializer#newInstance()} will be used instead
233   *
234   * @param consumer a {@link Consumer} of an {@link SeContainer}; may
235   * be {@code null}
236   *
237   * @param commandLineArguments the command line arguments; may be
238   * {@code null}
239   *
240   * @exception MalformedURLException if there was a problem
241   * formulating a classpath {@link URL}
242   * 
243   * @see org.microbean.main.Main
244   *
245   * @see SeContainerInitializer
246   *
247   * @see SeContainer
248   *
249   * @see <a
250   * href="https://ljnelson.github.io/microbean-maven-cdi/">MicroBean
251   * Maven CDI</a>
252   *
253   * @see <a
254   * href="https://ljnelson.github.io/microbean-main/">MicroBean
255   * Main</a>
256   *
257   * @see <a
258   * href="https://ljnelson.github.io/microbean-commons-cli/">MicroBean
259   * Commons CLI</a>
260   */
261  public static final void main(SeContainerInitializer bootstrapInitializer, SeContainerInitializer initializer, final Consumer<? super SeContainer> consumer, final String[] commandLineArguments) throws MalformedURLException {
262    if (bootstrapInitializer == null) {
263      bootstrapInitializer = SeContainerInitializer.newInstance();
264    }
265    assert bootstrapInitializer != null;
266    bootstrapInitializer.addExtensions(new MavenExtension());
267    bootstrapInitializer.addBeanClasses(ClasspathExporter.class, org.microbean.main.Main.class);
268
269    final ClasspathExporter[] exporterHolder = new ClasspathExporter[1];
270    org.microbean.main.Main.main(bootstrapInitializer, c -> exporterHolder[0] = c.select(ClasspathExporter.class).get(), commandLineArguments);
271    final ClasspathExporter exporter = exporterHolder[0];
272    assert exporter != null;
273    final Set<URI> classpathAdditions = exporter.getClasspath();
274    
275    if (initializer == null) {
276      initializer = SeContainerInitializer.newInstance();
277    }
278    final ClassLoader contextClassLoader = Thread.currentThread().getContextClassLoader();
279    URLClassLoader urlClassLoader = null;
280    if (classpathAdditions != null && !classpathAdditions.isEmpty()) {
281      final URL[] urls = new URL[classpathAdditions.size()];
282      int i = 0;
283      for (final URI uri : classpathAdditions) {
284        urls[i++] = uri.toURL();
285      }
286      urlClassLoader = doPrivileged((PrivilegedAction<URLClassLoader>)() -> new URLClassLoader(urls, contextClassLoader));
287      initializer.setClassLoader(urlClassLoader);
288    }
289    
290    try {
291      if (urlClassLoader != null) {      
292        Thread.currentThread().setContextClassLoader(urlClassLoader);
293      }
294      org.microbean.main.Main.main(initializer, null, exporter.getRemainingCommandLineArguments());
295    } finally {
296      Thread.currentThread().setContextClassLoader(contextClassLoader);
297    }
298  }
